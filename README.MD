# Z3_EASY

```PYTHON
from z3 import *

heso = [
    [2, 3, 4, 5],
    [12, 4, 6, 2],
    [3, 5, 2, 5],
    [2, 0, 3, 23]
]

ans = [
    14, 24, 15, 28
]

def hpt(word, index):
    res = 0
    for i in range(len(ans)): res += word[i] * heso[index][i]
    return res & 0xff

flag = [BitVec(f"flag[{i}]", 64) for i in range(4)]
s = Solver()

for i in range(len(ans)):
    s.add(ans[i] == hpt(flag, i))
print(s)
if s.check() == sat:
    m = s.model()
    for c in flag:
        print(m[c].as_long())
```
